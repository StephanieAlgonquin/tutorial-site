Building an Artist Tool

C# .NET 10 Windows Forms SQLite CSS

In this tutorial you'll build Image Board Studio — a desktop tool for artists to collect visual references and extract color palettes from images. You'll learn C#, Windows Forms, and SQLite by building a real multi-page app from scratch.

Windows Forms (WinForms) is Microsoft's framework for building Windows desktop apps. Each window is a Form, and UI elements — buttons, panels, labels — are Controls placed inside it.

ℹ️
This tutorial requires Windows 10 or 11. The app uses Windows-specific graphics APIs.

What You'll Build

The app has three screens:

  • Start Page — landing screen with two mode-selection cards
  • Image Board — freeform zoomable canvas; images drag, resize, and save to SQLite
  • Color Picker — load any image, click pixels, build a reusable color palette

Part 1 — Data Models

Create a Models folder, then add Models.cs inside it. These classes map directly to database rows.

1
Define Board and BoardItem
namespace Imageboard_illust.Models
{
    public class Board
    {
        public int    Id        { get; set; }
        public string Name      { get; set; } = "";
        public string CreatedAt { get; set; } = "";

        public string ImagesDir(string dataDir)
            => Path.Combine(dataDir, "boards", Id.ToString(), "images");
    }

    public class BoardItem
    {
        public int    Id, BoardId, X, Y, Width, Height, ZOrder;
        public string Name     { get; set; } = "";
        public string FileName { get; set; } = "";
    }
}

Part 2 — Database

Create a Database folder and add AppDatabase.cs. This class owns the SQLite connection for the app's lifetime.

1
Initialize the schema
using Microsoft.Data.Sqlite;

public class AppDatabase : IDisposable
{
    private readonly SqliteConnection _conn;
    public string DataDir { get; }

    public AppDatabase()
    {
        DataDir = Path.Combine(
            Environment.GetFolderPath(
                Environment.SpecialFolder.ApplicationData),
            "ImageBoardStudio");
        Directory.CreateDirectory(DataDir);
        _conn = new SqliteConnection(
            $"Data Source={Path.Combine(DataDir, "app.db")}");
        _conn.Open();
        InitSchema();
    }

    private void InitSchema()
    {
        Exec(@"CREATE TABLE IF NOT EXISTS boards (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            name TEXT NOT NULL,
            created_at TEXT DEFAULT (datetime('now'))
        );");
        Exec(@"CREATE TABLE IF NOT EXISTS board_items (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            board_id INTEGER NOT NULL,
            name TEXT, filename TEXT,
            x INT, y INT, w INT, h INT,
            z_order INT DEFAULT 0
        );");
    }

    public void Dispose() => _conn.Dispose();
}
2
Add read/write methods
public int InsertBoard(string name)
{
    using var cmd = _conn.CreateCommand();
    cmd.CommandText =
        "INSERT INTO boards (name) VALUES ($n); SELECT last_insert_rowid();";
    cmd.Parameters.AddWithValue("$n", name);
    return (int)(long)cmd.ExecuteScalar()!;
}

public List<Board> GetBoards()
{
    var list = new List<Board>();
    using var cmd = _conn.CreateCommand();
    cmd.CommandText =
        "SELECT id, name, created_at FROM boards ORDER BY id;";
    using var r = cmd.ExecuteReader();
    while (r.Read())
        list.Add(new Board {
            Id = r.GetInt32(0), Name = r.GetString(1),
            CreatedAt = r.GetString(2)
        });
    return list;
}

Paint the dot-grid background
Paint += (_, e) =>
{
    using var br = new LinearGradientBrush(
        ClientRectangle,
        Color.FromArgb(22, 22, 28),
        Color.FromArgb(30, 28, 38),
        LinearGradientMode.ForwardDiagonal);
    e.Graphics.FillRectangle(br, ClientRectangle);

    using var dot = new SolidBrush(Color.FromArgb(35, 255, 255, 255));
    const int gap = 32;
    for (int x = gap; x < Width;  x += gap)
    for (int y = gap; y < Height; y += gap)
        e.Graphics.FillEllipse(dot, x-1, y-1, 2, 2);
};
💡
Always call Invalidate() after changing a visual property drawn in a Paint handler — it schedules a repaint on the next UI cycle.

Part 4 — Image Board

The canvas is a scrollable Panel with a 4000×3000 virtual workspace. Images are draggable child controls.

1
Build the grid canvas
_canvas = new Panel {
    Dock = DockStyle.Fill,
    AutoScroll = true,
    AutoScrollMinSize = new Size(4000, 3000),
    BackColor = Color.FromArgb(40, 40, 43)
};
_canvas.Paint += (_, e) => {
    const int G = 40;
    using var pen = new Pen(Color.FromArgb(50, 50, 53), 1);
    int ox = _canvas.AutoScrollPosition.X % G;
    int oy = _canvas.AutoScrollPosition.Y % G;
    for (int x = ox; x < _canvas.Width; x += G)
        e.Graphics.DrawLine(pen, x, 0, x, _canvas.Height);
    for (int y = oy; y < _canvas.Height; y += G)
        e.Graphics.DrawLine(pen, 0, y, _canvas.Width, y);
};
2
Zoom with Ctrl + Scroll
_canvas.MouseWheel += (_, e) => {
    if (ModifierKeys.HasFlag(Keys.Control))
        SetZoom(_zoom + (e.Delta > 0 ? +0.1f : -0.1f));
};

private void SetZoom(float v)
{
    _zoom = Math.Clamp(v, 0.5f, 2.0f);
    foreach (var card in _cards.Values)
    {
        card.Location = new Point(
            (int)(card.Item.X * _zoom),
            (int)(card.Item.Y * _zoom));
        card.Size = new Size(
            (int)(card.Item.Width  * _zoom),
            (int)(card.Item.Height * _zoom));
    }
}

Part 5 — Color Picker

1
Read pixel color on hover
private void PicBox_MouseMove(object? s, MouseEventArgs e)
{
    if (_loadedImage == null) return;
    var bmp = (Bitmap)_loadedImage;

    // Convert display coords → actual bitmap coords
    int bx = (int)(e.X * (float)bmp.Width  / _picBox.Width);
    int by = (int)(e.Y * (float)bmp.Height / _picBox.Height);
    bx = Math.Clamp(bx, 0, bmp.Width  - 1);
    by = Math.Clamp(by, 0, bmp.Height - 1);

    Color c = bmp.GetPixel(bx, by);
    _hoveredSwatch.BackColor = c;
    _hoveredColorLbl.Text    = $"#{c.R:X2}{c.G:X2}{c.B:X2}";
}
⚠️
When SizeMode is Zoom, WinForms adds letterbox padding. Subtract the offset from e.X / e.Y for accurate edge picks.

Part 6 — Main Form

1
Page switching
private void ShowStartPage()
{
    ClearPages();
    var p = new StartPage();
    p.ImageBoardChosen  += (_, _) => ShowImageBoard();
    p.ColorPickerChosen += (_, _) => ShowColorPicker();
    Controls.Add(p);
}

private void ClearPages()
{
    foreach (Control c in Controls) c.Dispose();
    Controls.Clear();
}

protected override void OnFormClosed(FormClosedEventArgs e)
    { _db.Dispose(); base.OnFormClosed(e); }
